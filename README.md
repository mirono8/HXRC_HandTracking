# XR-Space Hand Tracking

JSON Names
- "eT" = elapsedTime
- "T" = task

- "iE" = interactableEvents
    - "e" = events
        - "sP" = startPosition
        - "eP" = endPosition
        - "iT" = interactableType ("b" = button, "s" = switch, "l" = lever)
        - "iS" = interactableSize ("s" = small, "m" = medium, "l" = large)
        - "di" = distance
        - "du" = duration
        - "tr" = trajectory
    
- "tS" = timeStamp
- "hP" - handPosition
- "cE" - collisionEvent
